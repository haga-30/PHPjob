SQL課題 ブログ用のデータベースを作成しよう

データベース作成
 create database YIGroupBlog;

テーブル作成（comments）
CREATE TABLE comments (
-> id INT NOT NULL AUTO_INCREMENT,
-> post_id INT,
-> name VARCHAR(20),
-> content VARCHAR(255),
-> time TIMESTAMP,
-> PRIMARY KEY (id)
-> )DEFAULT CHARSET=utf8;

テーブル作成（users）
CREATE TABLE users (
-> id INT AUTO_INCREMENT,
-> name VARCHAR(50),
-> password VARCHAR(255),
-> time TIMESTAMP,
-> PRIMARY KEY (id)
-> )DEFAULT CHARSET=utf8;
